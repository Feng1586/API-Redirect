## 二、数据库设计

### 2.1 MySQL 表结构

```sql
-- 用户表
CREATE TABLE users (
    id              BIGINT PRIMARY KEY AUTO_INCREMENT,
    email           VARCHAR(255) NOT NULL UNIQUE,
    username        VARCHAR(50) NOT NULL UNIQUE,
    password_hash   VARCHAR(255) NOT NULL,
    balance         DECIMAL(10, 4) DEFAULT 0.0000,     -- 余额
    role            ENUM('user', 'admin') DEFAULT 'user',
    is_active       TINYINT(1) DEFAULT 1,              -- 软删除标记
    created_at      DATETIME DEFAULT CURRENT_TIMESTAMP,
    updated_at      DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    deleted_at      DATETIME NULL,
    INDEX idx_email (email),
    INDEX idx_username (username)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- API-Key表
CREATE TABLE api_keys (
    id              BIGINT PRIMARY KEY AUTO_INCREMENT,
    user_id         BIGINT NOT NULL,
    key_name        VARCHAR(100) NOT NULL,              -- API-Key名称
    api_key         VARCHAR(50) NOT NULL UNIQUE,        -- 实际API-Key
    key_prefix      VARCHAR(20) NOT NULL,                -- 仅存储前8后4位
    created_at      DATETIME DEFAULT CURRENT_TIMESTAMP,
    last_used_at    DATETIME NULL,
    INDEX idx_user_id (user_id),
    INDEX idx_api_key (api_key),
    FOREIGN KEY (user_id) REFERENCES users(id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- 模型配置表
CREATE TABLE model_configs (
    id              BIGINT PRIMARY KEY AUTO_INCREMENT,
    model_name      VARCHAR(100) NOT NULL UNIQUE,       -- 模型名称
    price_per_1k    DECIMAL(10, 6) NOT NULL,            -- 每1000 Tokens价格
    is_enabled     TINYINT(1) DEFAULT 1,               -- 是否启用
    created_at      DATETIME DEFAULT CURRENT_TIMESTAMP,
    updated_at      DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- 上游配置表
CREATE TABLE upstream_config (
    id              BIGINT PRIMARY KEY AUTO_INCREMENT,
    config_key      VARCHAR(100) NOT NULL UNIQUE,
    config_value    TEXT NOT NULL,
    updated_at      DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;
-- 预计存储：upstream_api_key, upstream_base_url

-- 订单表
CREATE TABLE orders (
    id              BIGINT PRIMARY KEY AUTO_INCREMENT,
    order_no        VARCHAR(50) NOT NULL UNIQUE,        -- 订单编号
    user_id         BIGINT NOT NULL,
    amount          DECIMAL(10, 4) NOT NULL,            -- 充值金额
    status          ENUM('pending', 'paid', 'failed', 'refunded') DEFAULT 'pending',
    payment_method  VARCHAR(50) DEFAULT 'paypal',       -- 支付方式
    external_no     VARCHAR(255) NULL,                  -- 第三方订单号
    created_at      DATETIME DEFAULT CURRENT_TIMESTAMP,
    paid_at         DATETIME NULL,
    INDEX idx_user_id (user_id),
    INDEX idx_order_no (order_no),
    FOREIGN KEY (user_id) REFERENCES users(id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- 用量记录表
CREATE TABLE usage_records (
    id              BIGINT PRIMARY KEY AUTO_INCREMENT,
    user_id         BIGINT NOT NULL,
    api_key_id      BIGINT NOT NULL,
    model_name      VARCHAR(100) NOT NULL,
    prompt_tokens   INT NOT NULL DEFAULT 0,
    completion_tokens INT NOT NULL DEFAULT 0,
    total_tokens    INT NOT NULL DEFAULT 0,
    cost            DECIMAL(10, 6) NOT NULL,            -- 本次消费金额
    created_at      DATETIME DEFAULT CURRENT_TIMESTAMP,
    INDEX idx_user_date (user_id, created_at),
    INDEX idx_api_key (api_key_id),
    FOREIGN KEY (user_id) REFERENCES users(id),
    FOREIGN KEY (api_key_id) REFERENCES api_keys(id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;
```

# Session存储

TTL:    86400 (1天)

Key:    verify_code:{email}
Value:  { "code": "123456", "created_at": "...", "attempts": 0 }

# 验证码请求频率限制
Key:    rate_limit:verify:{email}
Value:  "1"

# 登录错误计数
Key:    login_fail:{email_or_username}
Value:  错误次数
TTL:    900 (15分钟)

# 用户当日用量缓存
Key:    daily_usage:{user_id}:{date}
Value:  JSON { "tokens": 1234, "cost": 0.05, "requests": 10 }
TTL:    86400 (1天)

### 5.7 认证服务 (services/auth_service.py)

```python
class AuthService:
    def send_register_code(self, email: str) -> tuple[bool, str]:
        """
        发送注册验证码
        检查: 邮箱是否已注册、请求频率限制
        """
  
    def register(self, email: str, code: str, username: str, password: str) -> tuple[bool, str, User]:
        """
        用户注册
        1. 验证验证码
        2. 验证密码强度
        3. 创建用户
        4. 创建Session
        5. 设置Cookie
        """
  
    def login(self, identifier: str, password: str) -> tuple[bool, str, User]:
        """
        用户登录
        1. 验证账号密码
        2. 检查登录错误次数
        3. 创建Session
        4. 设置Cookie
        """
  
    def logout(self, session_id: str) -> bool:
        """
        退出登录
        1. 删除Session
        2. 清除Cookie
        """
  
    def delete_account(self, user_id: int, email: str, code: str, password: str) -> tuple[bool, str]:
        """
        注销账户
        1. 验证验证码和密码
        2. 软删除用户
        3. 删除所有Session
        4. 发送通知邮件
        """
  
    def validate_session(self, session_id: str) -> User | None:
        """
        验证Session有效性
        有效则自动续期
        """
  
    def refresh_session(self, session_id: str) -> bool:
        """
        刷新Session有效期 (续期1天)
        """
```
